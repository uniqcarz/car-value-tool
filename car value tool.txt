<script>

function calculate(){

let price = document.getElementById("price").value;
let year = document.getElementById("year").value;
let km = document.getElementById("km").value;
let condition = document.getElementById("condition").value;

let currentYear = new Date().getFullYear();
let age = currentYear - year;

let value = price;

/* depreciation */

let depreciationRates = [0.15,0.10,0.10,0.10,0.05];

for(let i=0;i<age;i++){

if(i < depreciationRates.length)
value = value * (1 - depreciationRates[i]);

else
value = value * 0.95;

}

/* km adjustment */

if(km < 20000) value *= 1.05;
else if(km < 40000) value *= 1.02;
else if(km < 70000) value *= 1;
else if(km < 100000) value *= 0.95;
else value *= 0.90;

/* condition */

if(condition == "excellent") value *= 1.03;
if(condition == "fair") value *= 0.95;

value = Math.round(value);

document.getElementById("result").innerHTML =
"Estimated Car Value: ₹ " + value.toLocaleString();

}

</script>
