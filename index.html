let carDatabase=[]; async function loadCars(){ const response=await fetch("cars.json"); carDatabase=await response.json(); populateBrands(); } window.onload=loadCars; function populateBrands(){ let brandDropdown=document.getElementById("brand"); let brands=[...new Set(carDatabase.map(c=>c.brand))]; brands.forEach(b=>{ let opt=document.createElement("option"); opt.value=b; opt.text=b; brandDropdown.appendChild(opt); }); } function populateModels(){ let brand=document.getElementById("brand").value; let dropdown=document.getElementById("model"); dropdown.innerHTML=""; let models=carDatabase.filter(c=>c.brand==brand); let unique=[...new Set(models.map(c=>c.model))]; unique.forEach(m=>{ let opt=document.createElement("option"); opt.value=m; opt.text=m; dropdown.appendChild(opt); }); } function calculateValue(){ let brand=document.getElementById("brand").value; let model=document.getElementById("model").value; let year=document.getElementById("year").value; let km=document.getElementById("kms").value; let car=carDatabase.find(c=>c.brand==brand && c.model==model); if(!car){ document.getElementById("result").innerHTML="Car not found"; return; } let age=new Date().getFullYear()-year; let value=car.onroad*Math.pow(0.9,age); value*=car.bodyfactor; value*=car.fuelfactor; value*=car.transfactor; value*=car.ownerfactor; if(km>80000) value*=0.9; let min=Math.round(value*0.95); let max=Math.round(value*1.05); document.getElementById("result").innerHTML="₹"+min+" - ₹"+max; }
